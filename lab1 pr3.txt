#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <string>
#include <iostream>

int main()
{
    char str[50], * p, * array[100], * aux;
    int i = 0;
    printf("Insert the senctence here:");
    scanf_s("%[^\n]", str, sizeof(str));
    p = strtok(str, " ");

    while (p != NULL)
    {
        array[i] = p;
        //printf("%s.", array[i]);
        i++;
        p = strtok(NULL, " ");

    }
    for (int j = 0; j < i - 1; j++)
    {
        for (int k = j + 1; k < i; k++)
        {
            if ((strlen(array[j])) < (strlen(array[k])))
            {
                aux = array[j];
                array[j] = array[k];
                array[k] = aux;
            }
            else if ((strlen(array[j])) == (strlen(array[k])))
                if (strcmp(array[j], array[k]) > 0)
                {
                    aux = array[j];
                    array[j] = array[k];
                    array[k] = aux;
                }

        }

    }
    for (int f = 0; f < i; f++)
    {
        printf("%s\n", array[f]);
    }
    return 0;
}
